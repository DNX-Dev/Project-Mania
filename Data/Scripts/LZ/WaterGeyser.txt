event ObjectUpdate
	switch 4
	case 0
		object.value1 = object.ypos
		object.value2 = object.ypos
		object.value2 -= 0x1000000
		object.value3 = object.ypos
		object.value3 -= 0x900000
		if object.propertyValue == 0
			object.state++
		else
			object.value4 = 0xFFFFFF
			object.state = 3
		end if
		object.yvel = -0x100000
		break
	case 1
		object.yvel += 0x3800
		object.ypos += object.yvel
		if object.ypos <= object.value2
			object.ypos = object.value2
			object.state++
		end if
		break
	case 2
		object.yvel += 0x3800
		if object.yvel > 0
			object.ypos += object.yvel
		end if
		if object.ypos >= object.value1
			object.type = TypeName[Blank Object]
		end if
		break
	case 3
		object.yvel += 0x3800
		if object.yvel > 0
			object.ypos += object.yvel
			if object.value4 != 0xFFFFFF
				arrayPos0 = object.value4
				if object[arrayPos0].state != 13
					object[arrayPos0].state = 11
				end if
				object[arrayPos0].yvel = 0
				object[arrayPos0].value13 = 0
				object.value4 = 0xFFFFFF
			end if
		else
			if object.value4 != 0xFFFFFF
				foreach (TypeName[B08 Eggman], arrayPos0, ACTIVE_ENTITIES)
					BoxCollisionTest(C_TOUCH, object.entityPos, -14, -18, 14, 18, arrayPos0, -12, -24, 12, 24)
					if checkResult == 1
						object.value2 = object.value3
						object.value4 = arrayPos0
						if object[arrayPos0].state != 13
							object[arrayPos0].state = 10
						end if
					end if
				next
			end if
		end if
		if object.ypos >= object.value1
			object.type = TypeName[Blank Object]
		end if
		break
	end switch
	
	foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
		BoxCollisionTest(C_TOUCH, object.entityPos, -2, -2, 2, 2, currentPlayer, C_BOX, C_BOX, C_BOX, C_BOX)
		
		if checkResult == true
		
			player.score = 69
			object[currentPlayer].state = Player_State_Air
			object[currentPlayer].gravity = GRAVITY_AIR
			object[currentPlayer].animation = ANI_WATERSLIDE
			object[currentPlayer].yvel = -0x0B0000
			object[currentPlayer].speed = 0
			object[currentPlayer].xvel = 0
			object[currentPlayer].value1 = 0
			if object.state == 0
				object.state = 1
				CreateTempObject(TypeName[Water Geyser], 0, object.xpos, object.ypos)
				PlaySfx(SfxName[Water Geyser], 0)
			end if
		end if
	next
	if object.state == 1
		object.value0++
		if object.value0 == 64
			object.value0 = 0
			object.state = 0
		end if
		StopSfx(SfxName[Water Splash])
	end if
end event
